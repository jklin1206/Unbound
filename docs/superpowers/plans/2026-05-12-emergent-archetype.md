**Goal:** Replace the chosen-at-onboarding `Archetype` enum with `BuildIdentity` — a grounded athletic descriptor auto-derived from `AttributeProfile`. Migrate all 12 consumers across cinematic, report, body analysis, program generation, badges, and onboarding; delete the enum at the end.

Vocabulary table per axis (Power/Movement/Control/Endurance/Mobility/Explosive) × 5 distribution shapes (balancedAthlete / hybridAthlete / specialist / hybrid / lean) → 20 distinct displayName outputs. Existing `AttributeProfile.buildName: String` stays as a backward-compat alias during migration phases.

**Tech Stack:** Swift 5.9+, SwiftUI iOS 17+, XCTest. Tests run via `xcodebuild test`. Project regenerated by xcodegen.

### Task 2a.2: BuildIdentity struct + Shape enum

**Files:**

- [ ] **Step 1: Write the implementation**

```swift
// UNBOUND/Models/BuildIdentity.swift
import Foundation

// MARK: - BuildIdentity
//
// Grounded athletic descriptor derived from AttributeProfile. Replaces the
// chosen-at-onboarding Archetype enum.
//
// Two-layer rule (per feedback_unbound_buildidentity_vs_titles memory):
//   BuildIdentity → grounded (auto-derived).
//   Title         → earned fantasy flavor (separate future system).

struct BuildIdentity: Equatable, Sendable {
    enum Shape: String, Codable, Sendable {
        case balancedAthlete       // spread < 15
        case hybridAthlete          // top3 within 10 of top1 AND gap12 < 10
        case specialist             // gap12 > 25
        case hybrid                 // gap12 < 10 (and NOT hybridAthlete)
        case lean                   // 10 ≤ gap12 ≤ 25
    }

    /// Primary axis — nil for balancedAthlete and hybridAthlete.
    let primary: AttributeKey?
    /// Secondary axis — only set for .hybrid.
    let secondary: AttributeKey?
    let shape: Shape

    var displayName: String {
        switch shape {
        case .balancedAthlete: return "Balanced Athlete"
        case .hybridAthlete:   return "Hybrid Athlete"
        case .specialist:
            guard let primary else { return "Balanced Athlete" }
            return "\(primary.buildVocab) Specialist"
        case .hybrid:
            guard let primary else { return "Balanced Athlete" }
            return "\(primary.buildVocab) Hybrid"
        case .lean:
            guard let primary else { return "Balanced Athlete" }
            return "\(primary.buildVocab)\(primary.leanSuffix)"
        }
    }

    var tagline: String {
        switch shape {
        case .balancedAthlete:
            return "Even across every axis."
        case .hybridAthlete:
            return "Multi-axis athlete — no single specialty."
        case .specialist:
            guard let primary else { return "Even across every axis." }
            return "Built around \(primary.taglinePhrase) — sharply focused."
        case .hybrid:
            guard let primary, let secondary else { return "Even across every axis." }
            return "Built around \(primary.taglinePhrase) with strong \(secondary.buildVocab)."
        case .lean:
            guard let primary else { return "Even across every axis." }
            return "Trending toward \(primary.taglinePhrase)."
        }
    }
}
```

    /// Aggregate sub-rank across the user's BuildIdentity primary axis (or
    /// top-3 axes for balanced/hybridAthlete). Replaces archetypeRank.
    func aggregateRank(userId: String) async -> SubRank {
        // Load attribute profile to derive BuildIdentity
        let profile = ServiceContainer.shared.attribute.snapshot(userId: userId, asOf: .now)
        let identity = profile.buildIdentity
        let lifts: [String]
        switch identity.shape {
        case .balancedAthlete, .hybridAthlete:
            // Top-3 axes by peak, union of their emphasis lifts
            let sorted = AttributeKey.allCases
                .sorted { profile.value(for: $0).peak > profile.value(for: $1).peak }
            lifts = Array(sorted.prefix(3).flatMap { $0.emphasisLifts })
        case .specialist, .hybrid, .lean:
            lifts = identity.primary?.emphasisLifts ?? []
        }
        guard !lifts.isEmpty else { return .eMinus }
        // Reuse the existing aggregate-over-lifts logic. Pull the user's
        // per-lift ranks and average them.
        let all = await fetchAll(userId: userId)
        let matched = all.filter { lifts.contains($0.exerciseKey) }
        guard !matched.isEmpty else { return .eMinus }
        let avgOrdinal = matched.map { $0.currentRank.ordinal }.reduce(0, +) / matched.count
        return SubRank.allCases.first(where: { $0.ordinal == avgOrdinal }) ?? .eMinus
    }
